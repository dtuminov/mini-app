### install nginx
sudo apt install nginx-core

### edit cfg
sudo nano /etc/nginx/sites-available/default

### cfg
server {
	listen 443 ssl;
	server_name cardsasigsuadkh.ru;

	location /ghjfkyui6 {
		proxy_set_header Host $http_host;
		proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
		proxy_redirect off;
		proxy_buffering off;
		proxy_pass http://127.0.0.1:8080;
	}

	ssl_certificate /etc/letsencrypt/live/cardsasigsuadkh.ru/fullchain.pem;
	ssl_certificate_key /etc/letsencrypt/live/cardsasigsuadkh.ru/privkey.pem;
}

### check cfg
sudo nginx -t

### reload service
sudo systemctl restart nginx.service
